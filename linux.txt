ls 
	-a //显示隐藏文件
	-l //列表形式显示
	-h //人性化显示（配合-l使用

cat //查看文件内容

pwd //查看当前文件路径

chmod //修改权限
    u 表示“用户（user）”，即文件或目录的所有者。
　　g 表示“同组（group）用户”，即与文件属主有相同组ID的所有用户。
　　o 表示“其他（others）用户”
    r       read          4
    w       wirte         2
    x       execute       1
    -                     0
    文件类型:
    d   目录
    c   字符串文件
    l   链接文件


rm 
	-r //删除文件夹
	-f //不确认

who //查看当前登录系统的用户,终端,时间

date //打印当前时间

ps //显示当前进程，默认只显示当前用户创建的进程
	-e	//所有进程

kill //杀死某个进程

uname //显示机器名称
	-a //完整名称
